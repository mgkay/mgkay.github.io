(2b) Since addition cube capacity is available at an additional charge, what is the minimum increase in the cubic feet of items that can be packed that would maximize total profit?

In [3]:
function cube2profit(k)
    model = Model(GLPK.Optimizer)
    @variable(model, 0 <= x[i = N] <= u[i] )
    @objective(model, Max, sum((p[i] - c[i])*x[i] for i ∈ N) )
    @constraint(model, sum(c[i]*x[i] for i ∈ N) <= 120 )
    @constraint(model, sum(x[i] for i ∈ N) <= 30 )
    @constraint(model, sum(x[i]/s[i] for i ∈ N) <= k )
    optimize!(model)
    return objective_value(model)
end
k = [0:.1:5;]    # Cube capacity

using Base    # Needed for diff function

function bmax2nogen(bmax)
    bmin = min.(b,bmax)    # Take min of b and bmax for each month
    model = Model(GLPK.Optimizer)
    @variable(model, 0 <= x[1:12], Int)
    @objective(model, Min, sum(x[i] for i = 1:12) )
    for i = 1:12
        @constraint(model, sum(x[Int(j - 12*floor((j - 1)/12))] for j = i+4:i+12) >= bmin[i]/25 )
    end
    optimize!(model)
    return nᵒ = objective_value(model)
end
