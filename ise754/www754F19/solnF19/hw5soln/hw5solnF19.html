EXAMPLE 4: UFL with n,m = 104

clear all
x = uszip5(mor({'NC'},uszip5('ST')) & uszip5('Pop') > 30000);
P = x.XY;
a = x.LandArea;
dafh = @(XY1,a1,XY2,a2) max(1.2*dists(XY1,XY2,'mi'),...
   0.675*max(sqrt(a1(:)),sqrt(a2(:)')));

f = x.Pop';                 % person
r = 1/10000;                % $/person-mi
